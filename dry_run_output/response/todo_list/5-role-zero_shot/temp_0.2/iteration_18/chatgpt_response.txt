```python
"""
Console-based Todo List Application adhering to ISO/IEC 25010 requirements.

This application provides a TaskManager class for managing tasks. 
Each function includes validation, error handling, and efficient in-memory operations.

Author: AI Assistant
"""

from typing import List, Dict

class TaskValidationError(ValueError):
    """Custom error for invalid task operations."""
    pass

class TaskNotFoundError(KeyError):
    """Custom error for cases where task is not found."""
    pass

class TaskManager:
    """
    Manages todo tasks with add, remove, search, finish, get_all, clear_all functionalities.
    """
    def __init__(self):
        # Internal task storage: {task_id: {'id', 'task_name', 'task_description', 'is_finished'}}
        self._tasks: Dict[int, Dict] = {}
        self._next_id: int = 1

    def add(self, task_name: str, task_description: str) -> int:
        """
        Adds a new task to the manager.
        
        Args:
            task_name (str): The task's name (non-empty).
            task_description (str): The task's description (non-empty).
        
        Returns:
            int: The unique ID of the created task.
        
        Raises:
            ValueError: If task_name or task_description is empty or whitespace.
        """
        if not isinstance(task_name, str) or not task_name.strip():
            raise ValueError("Task name must be a non-empty string.")
        if not isinstance(task_description, str) or not task_description.strip():
            raise ValueError("Task description must be a non-empty string.")

        task_id = self._next_id
        self._tasks[task_id] = {
            "id": task_id,
            "task_name": task_name.strip(),
            "task_description": task_description.strip(),
            "is_finished": False
        }
        self._next_id += 1
        return task_id

    def remove(self, task_id: int) -> bool:
        """
        Removes a task from the manager by its ID.

        Args:
            task_id (int): The ID of the task to remove.
        
        Returns:
            bool: True if the task was removed successfully.
        
        Raises:
            ValueError: If task_id is not a positive integer.
            TaskNotFoundError: If task does not exist.
        """
        if not isinstance(task_id, int) or task_id <= 0:
            raise ValueError("Task ID must be a positive integer.")
        if task_id not in self._tasks:
            raise TaskNotFoundError(f"Task with ID {task_id} not found.")
        del self._tasks[task_id]
        return True

    def search(self, task_term: str) -> List[Dict]:
        """
        Searches tasks by name or description (case-insensitive substring match).

        Args:
            task_term (str): The search term (non-empty string).
        
        Returns:
            list[dict]: List of matching tasks in standard representation.
        
        Raises:
            ValueError: If task_term is empty.
        """
        if not isinstance(task_term, str) or not task_term.strip():
            raise ValueError("Search term must be a non-empty string.")

        term = task_term.strip().lower()
        results = [
            self._format_task(task)
            for task in self._tasks.values()
            if term in task["task_name"].lower() or term in task["task_description"].lower()
        ]
        return results

    def finish(self, task_id: int) -> bool:
        """
        Marks a task as completed.

        Args:
            task_id (int): The ID of the task to mark as finished.
        
        Returns:
            bool: True if the task was marked as finished.
        
        Raises:
            ValueError: If task_id is not a positive integer.
            TaskNotFoundError: If task does not exist.
        """
        if not isinstance(task_id, int) or task_id <= 0:
            raise ValueError("Task ID must be a positive integer.")
        if task_id not in self._tasks:
            raise TaskNotFoundError(f"Task with ID {task_id} not found.")
        self._tasks[task_id]["is_finished"] = True
        return True

    def get_all(self) -> List[Dict]:
        """
        Retrieves all tasks.

        Returns:
            list[dict]: All tasks in standard representation.
        """
        return [self._format_task(task) for task in self._tasks.values()]

    def clear_all(self) -> bool:
        """
        Deletes all tasks.

        Returns:
            bool: True if tasks successfully cleared.
        """
        self._tasks.clear()
        return True

    @staticmethod
    def _format_task(task: Dict) -> Dict:
        """
        Formats a single task dict in the required structure.

        Args:
            task (dict): Task dictionary.
        Returns:
            dict: Formatted task dictionary.
        """
        return {
            "id": task["id"],
            "task_name": task["task_name"],
            "task_description": task["task_description"],
            "is_finished": task["is_finished"]
        }

if __name__ == "__main__":
    # Example usage and minimal test cases for demonstration
    tm = TaskManager()

    # Add tasks
    task1_id = tm.add("Buy groceries", "Milk, eggs, bread")
    task2_id = tm.add("Work out", "Cardio for 30 minutes")

    # Finish a task
    tm.finish(task1_id)

    # Get all
    print("All tasks:", tm.get_all())

    # Search for a task
    print("Search 'work':", tm.search("work"))

    # Remove a task
    tm.remove(task1_id)
    print("After removing task 1:", tm.get_all())

    # Clear all tasks
    tm.clear_all()
    print("After clearing:", tm.get_all())
```
**Key Points:**
- All operations validate inputs and raise precise exceptions.
- Optimized for lookups by using a dictionary for O(1) operations.
- Tasks are represented with all required fields: `(id, task_name, task_description, is_finished)`.
- Output is easy to test and extend.
- Thorough docstrings and logical code organization facilitate maintainability and clarity.
