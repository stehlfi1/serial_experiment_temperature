```python
import threading
from typing import List, Dict, Optional

class TaskManager:
    """
    A high-quality, maintainable, console-based todo list manager.

    Features (as per ISO/IEC 25010):
        - Functional correctness and reliability.
        - Data and interface validation.
        - Performance with O(1) operations for add/remove/get by ID.
        - Encapsulation and modularity for maintainability and testability.
        - Thread-safety for predictable modifications.
    """

    class _Task:
        """
        Internal Task representation.
        """
        def __init__(self, task_id: int, name: str, description: str):
            self.id = task_id
            self.task_name = name
            self.task_description = description
            self.is_finished = False

        def to_dict(self) -> Dict:
            return {
                "id": self.id,
                "task_name": self.task_name,
                "task_description": self.task_description,
                "is_finished": self.is_finished
            }

    def __init__(self) -> None:
        """
        Initializes the TaskManager with empty task storage and unique ID generator.
        """
        self._tasks: Dict[int, TaskManager._Task] = {}
        self._next_id: int = 1                       # Invariant: always next unused ID
        self._lock = threading.Lock()                # Thread safe for future extension

    def add(self, task_name: str, task_description: str) -> int:
        """
        Adds a new task to the todo list.

        Args:
            task_name (str): Name of the task (non-empty)
            task_description (str): Description of the task (non-empty)

        Returns:
            int: The unique ID of the created task.

        Raises:
            ValueError: If task_name or task_description is empty.
        """
        if not isinstance(task_name, str) or not task_name.strip():
            raise ValueError("Task name must be a non-empty string")
        if not isinstance(task_description, str) or not task_description.strip():
            raise ValueError("Task description must be a non-empty string")

        with self._lock:
            task_id = self._next_id
            self._tasks[task_id] = TaskManager._Task(task_id, task_name.strip(), task_description.strip())
            self._next_id += 1
            return task_id

    def remove(self, task_id: int) -> bool:
        """
        Removes a task by its ID.

        Args:
            task_id (int): The ID of the task to remove.

        Returns:
            bool: True if task was removed, False if not found / invalid input.

        Raises:
            ValueError: If task_id is not a valid positive integer.
        """
        if not isinstance(task_id, int) or task_id < 1:
            raise ValueError("Task ID must be a positive integer")
        with self._lock:
            return self._tasks.pop(task_id, None) is not None

    def search(self, task_term: str) -> List[Dict]:
        """
        Searches for tasks by name or description (case-insensitive substring match).

        Args:
            task_term (str): Search term (non-empty).

        Returns:
            List[Dict]: List of all tasks matching the search term.

        Raises:
            ValueError: If task_term is empty.
        """
        if not isinstance(task_term, str) or not task_term.strip():
            raise ValueError("Search term must be a non-empty string")

        term_lower = task_term.strip().lower()

        with self._lock:
            return [
                task.to_dict()
                for task in self._tasks.values()
                if term_lower in task.task_name.lower() or term_lower in task.task_description.lower()
            ]

    def finish(self, task_id: int) -> bool:
        """
        Marks a given task as finished.

        Args:
            task_id (int): The ID of the task to mark as finished.

        Returns:
            bool: True if successful, False if the task does not exist.

        Raises:
            ValueError: If task_id is not valid.
        """
        if not isinstance(task_id, int) or task_id < 1:
            raise ValueError("Task ID must be a positive integer")
        with self._lock:
            task = self._tasks.get(task_id)
            if task is not None:
                task.is_finished = True
                return True
            else:
                return False

    def get_all(self) -> List[Dict]:
        """
        Returns all tasks in the system.

        Returns:
            List[Dict]: List of all tasks, each with (id, task_name, task_description, is_finished).
        """
        with self._lock:
            return [task.to_dict() for task in self._tasks.values()]

    def clear_all(self) -> bool:
        """
        Deletes all tasks from the todo list.

        Returns:
            bool: True if tasks were deleted (or already empty).
        """
        with self._lock:
            self._tasks.clear()
            self._next_id = 1
            return True

# Example usage (uncomment for testing)
if __name__ == "__main__":
    tm = TaskManager()
    # Add tasks
    id1 = tm.add("Buy milk", "Get two liters of milk")
    id2 = tm.add("Read book", "Finish chapter 3 of the novel")
    # Search
    print(tm.search("milk"))                 # [{'id': 1, ...}]
    print(tm.get_all())                      # Returns all tasks
    print(tm.finish(id1))                    # True
    print(tm.remove(id2))                    # True
    print(tm.get_all())                      # Only id1
    print(tm.clear_all())                    # True
    print(tm.get_all())                      # []
```
