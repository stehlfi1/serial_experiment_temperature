```python
"""
Todo List Console Application

This code defines a high-quality, maintainable, and testable Python class named TaskManager that provides
a console-based todo list application, conforming to ISO/IEC 25010 software quality requirements.

Features:
- OOP design
- Input validation and robust error handling
- Efficient in-memory task storage
- Easily testable and extensible
"""

from typing import List, Dict


class TaskValidationError(ValueError):
    """Custom exception for invalid task operations."""
    pass


class TaskNotFoundError(LookupError):
    """Custom exception for non-existent task IDs."""
    pass


class Task:
    """
    Data structure representing a Task object.
    """
    __slots__ = ['id', 'name', 'description', 'is_finished']

    def __init__(self, id: int, name: str, description: str):
        self.id = id
        self.name = name
        self.description = description
        self.is_finished = False

    def to_dict(self) -> Dict:
        """
        Export task as a dictionary with complete information.
        """
        return {
            "id": self.id,
            "task_name": self.name,
            "task_description": self.description,
            "is_finished": self.is_finished
        }


class TaskManager:
    """
    TaskManager: Manages a todo list in-memory with efficient operations.
    """

    def __init__(self):
        """
        Initialize the TaskManager with empty storage and a unique ID counter.
        """
        self._tasks: Dict[int, Task] = {}   # Task storage: {id: Task}
        self._next_id: int = 1              # Monotonically increasing unique task id

    def _validate_task_name(self, name: str):
        if not isinstance(name, str) or not name.strip():
            raise TaskValidationError("Task name must be a non-empty string.")

    def _validate_task_description(self, description: str):
        if not isinstance(description, str) or not description.strip():
            raise TaskValidationError("Task description must be a non-empty string.")

    def _validate_task_id(self, task_id: int):
        if not isinstance(task_id, int) or task_id < 1:
            raise TaskValidationError("Task ID must be a positive integer.")

    def _get_task(self, task_id: int) -> Task:
        """
        Fetch a task by ID or raise if it does not exist.
        """
        if task_id not in self._tasks:
            raise TaskNotFoundError(f"Task with ID {task_id} does not exist.")
        return self._tasks[task_id]

    def add(self, task_name: str, task_description: str) -> int:
        """
        Adds a new task to the todo list.
        Args:
            task_name: Name/title of the task (non-empty string)
            task_description: Description of the task (non-empty string)
        Returns:
            The unique integer ID assigned to the newly created task
        Raises:
            TaskValidationError: On invalid input
        """
        self._validate_task_name(task_name)
        self._validate_task_description(task_description)

        task_id = self._next_id
        task = Task(task_id, task_name.strip(), task_description.strip())
        self._tasks[task_id] = task
        self._next_id += 1

        return task_id

    def remove(self, task_id: int) -> bool:
        """
        Remove the task with the given ID.
        Returns True if removed, or raises if not found.
        """
        self._validate_task_id(task_id)
        try:
            del self._tasks[task_id]
            return True
        except KeyError:
            raise TaskNotFoundError(f"Task with ID {task_id} does not exist.")

    def search(self, task_term: str) -> List[Dict]:
        """
        Search for tasks containing the search term (case-insensitive) in the task name or description.
        Returns a list of matching task dicts.
        If the search term is empty or not a string, raises error.
        """
        if not isinstance(task_term, str) or not task_term.strip():
            raise TaskValidationError("Search term must be a non-empty string.")
        term = task_term.strip().lower()
        matches = [
            task.to_dict()
            for task in self._tasks.values()
            if term in task.name.lower() or term in task.description.lower()
        ]
        return matches

    def finish(self, task_id: int) -> bool:
        """
        Mark the task as completed (is_finished = True).
        Returns True if successful, or raises if not found.
        """
        self._validate_task_id(task_id)
        task = self._get_task(task_id)
        if not task.is_finished:
            task.is_finished = True
        return True

    def get_all(self) -> List[Dict]:
        """
        Retrieve all tasks, each as a dictionary (id, task_name, task_description, is_finished).
        Returns an empty list if no tasks exist.
        """
        return [task.to_dict() for task in self._tasks.values()]

    def clear_all(self) -> bool:
        """
        Remove all tasks from the manager.
        Returns True if successful.
        """
        self._tasks.clear()
        return True


# Example usage (for demonstration/testing)
if __name__ == "__main__":
    # Instantiate TaskManager for demonstration
    manager = TaskManager()

    try:
        # Adding tasks
        task1_id = manager.add("Finish documentation", "Complete the project documentation before Friday.")
        task2_id = manager.add("Code review", "Review Alice's code for module X.")
        # Finishing a task
        manager.finish(task1_id)
        # Searching for a task
        print("Search 'review': ", manager.search("review"))
        # List all tasks
        print("All tasks: ", manager.get_all())
        # Remove a task
        manager.remove(task2_id)
        print("All tasks after removal: ", manager.get_all())
        # Clear all tasks
        manager.clear_all()
        print("All tasks after clearing: ", manager.get_all())
    except (TaskValidationError, TaskNotFoundError) as e:
        print("Error:", e)
```
**Key Notes:**

- **Efficient lookups**: Tasks are stored in a dict (`self._tasks`) with ID as key for O(1) access.
- **Validation**: Input is validated using helper methods, raising specific error types for clarity and safety.
- **Error Handling**: Uses custom exception classes to distinguish between validation and missing item errors.
- **Testability**: TaskManager methods can be directly unit tested; main block provides example usage.
- **Maintainability**: The structure is highly modular and documented; each method has clear purpose and error handling.
- **ISO/IEC 25010 alignment**: The code addresses usability (clear errors), reliability, efficiency, and maintainability.
