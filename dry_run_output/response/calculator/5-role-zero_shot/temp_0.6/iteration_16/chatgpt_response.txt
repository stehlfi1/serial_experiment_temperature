from typing import List, Tuple
import re

class CalculatorError(Exception):
    """Custom exception for calculator errors."""
    pass

class Calculator:
    """
    Console-based arithmetic calculator supporting +, -, *, /, (), correct precedence,
    and both integer and floating point numbers (positive and negative).
    """

    # Supported operators with their precedence and associativity
    _OPERATORS = {
        '+': (1, 'L'),
        '-': (1, 'L'),
        '*': (2, 'L'),
        '/': (2, 'L'),
    }

    # Top-level method as per interface
    def calculate(self, expression: str) -> float:
        """
        Parse, validate, and evaluate an arithmetic expression.

        Args:
            expression (str): The arithmetic expression as a string.

        Returns:
            float: The result of the evaluated expression.

        Raises:
            ValueError, ZeroDivisionError, CalculatorError: On invalid input or operation.
        """
        tokens = self._tokenize(expression)
        self._validate_tokens(tokens)
        rpn = self._to_rpn(tokens)
        result = self._evaluate_rpn(rpn)
        return result

    def _tokenize(self, expression: str) -> List[str]:
        """
        Tokenize the input expression string into numbers, operators, and parentheses.

        Args:
            expression (str): The arithmetic expression.

        Returns:
            List[str]: List of token strings.

        Raises:
            ValueError: For invalid characters or patterns.
        """
        # Tokenize numbers (int/float/negatives), operators, and parenthesis
        token_pattern = re.compile(r"""
            (?P<number>   -?\d+(\.\d+)? ) |  # Integer or decimal, possibly negative
            (?P<lparen>   \()           |    # Left parenthesis
            (?P<rparen>   \))           |    # Right parenthesis
            (?P<operator> [\+\-\*/])    |    # Arithmetic operators
            (?P<space>    \s+)               # Space (will be skipped)
        """, re.VERBOSE)

        tokens = []
        pos = 0
        last_token = None
        while pos < len(expression):
            match = token_pattern.match(expression, pos)
            if not match:
                raise ValueError(f"Invalid character at position {pos}: '{expression[pos]}'")

            if match.lastgroup == "number":
                tokens.append(match.group('number'))
                last_token = 'number'
            elif match.lastgroup == "operator":
                # To distinguish unary minus: if previous token is None or an operator or left parenthesis
                op = match.group('operator')
                if (op == '-' and
                        (last_token is None or last_token in ('operator', 'lparen'))):
                    # Handle unary minus (part of number)
                    num_match = token_pattern.match(expression, match.end())
                    if num_match and num_match.lastgroup == 'number':
                        tokens.append('-' + num_match.group('number').lstrip('-+'))
                        pos = num_match.end()  # consume the number token after the minus
                        last_token = 'number'
                        continue
                    else:
                        # Standalone unary minus (e.g., "-(3+4)")
                        tokens.append('-1')
                        tokens.append('*')
                        last_token = 'operator'
                else:
                    tokens.append(op)
                    last_token = 'operator'
            elif match.lastgroup == "lparen":
                tokens.append('(')
                last_token = 'lparen'
            elif match.lastgroup == "rparen":
                tokens.append(')')
                last_token = 'rparen'
            # spaces are ignored
            pos = match.end()
        return tokens

    def _validate_tokens(self, tokens: List[str]) -> None:
        """
        Validates the list of tokens for:
        - Only valid characters
        - Balanced parentheses

        Args:
            tokens (List[str]): Tokens from the input.

        Raises:
            ValueError: If expression is not valid.
        """
        parentheses = 0
        prev_token = None
        for t in tokens:
            if t == '(':
                parentheses += 1
            elif t == ')':
                parentheses -= 1
                if parentheses < 0:
                    raise ValueError("Unbalanced parentheses: too many ')'")
            elif t in self._OPERATORS:
                if prev_token is None or prev_token in self._OPERATORS or prev_token == '(':
                    raise ValueError(f"Operator '{t}' used in invalid context.")
            else:
                # should be a valid number
                try:
                    float(t)
                except ValueError:
                    raise ValueError(f"Invalid token: '{t}'")
            prev_token = t
        if parentheses != 0:
            raise ValueError("Unbalanced parentheses: too many '('")

    def _to_rpn(self, tokens: List[str]) -> List[str]:
        """
        Convert infix tokens to Reverse Polish Notation (Shunting Yard Algorithm).

        Args:
            tokens (List[str]): Infix tokens.

        Returns:
            List[str]: RPN token list.
        """
        output = []
        stack = []
        for token in tokens:
            if self._is_number(token):
                output.append(token)
            elif token in self._OPERATORS:
                while (stack and stack[-1] in self._OPERATORS and
                       ((self._OPERATORS[token][1] == 'L' and self._OPERATORS[token][0] <= self._OPERATORS[stack[-1]][0]) or
                        (self._OPERATORS[token][1] == 'R' and self._OPERATORS[token][0] < self._OPERATORS[stack[-1]][0]))):
                    output.append(stack.pop())
                stack.append(token)
            elif token == '(':
                stack.append(token)
            elif token == ')':
                while stack and stack[-1] != '(':
                    output.append(stack.pop())
                if not stack:
                    raise ValueError("Unbalanced parentheses: ')' without matching '('")
                stack.pop()  # Remove '('
        while stack:
            if stack[-1] in '()':
                raise ValueError("Unbalanced parentheses in expression.")
            output.append(stack.pop())
        return output

    def _evaluate_rpn(self, tokens: List[str]) -> float:
        """
        Evaluate an expression in Reverse Polish Notation.

        Args:
            tokens (List[str]): RPN token list.

        Returns:
            float: Evaluation result.

        Raises:
            ZeroDivisionError: if division by zero occurs.
            CalculatorError: if a malformed expression is detected.
        """
        stack = []
        for token in tokens:
            if self._is_number(token):
                stack.append(float(token))
            elif token in self._OPERATORS:
                if len(stack) < 2:
                    raise CalculatorError("Malformed expression: insufficient operands.")
                b = stack.pop()
                a = stack.pop()
                result = None
                if token == '+':
                    result = a + b
                elif token == '-':
                    result = a - b
                elif token == '*':
                    result = a * b
                elif token == '/':
                    if b == 0:
                        raise ZeroDivisionError("Division by zero is not allowed.")
                    result = a / b
                stack.append(result)
            else:
                raise CalculatorError(f"Invalid token in RPN: {token}")
        if len(stack) != 1:
            raise CalculatorError("Malformed expression.")
        return stack[0]

    @staticmethod
    def _is_number(token: str) -> bool:
        """Checks if the token represents a number (int/float)."""
        try:
            float(token)
            return True
        except ValueError:
            return False

    calc = Calculator()

        "1 + 2 * 3",                      # 7.0
        "(1 + 2) * 3",                    # 9.0
        "-4 + 2",                         # -2.0
        "3.5 * 2 - (7 / -2)",             # 10.5
        "-(3 + 4) * 2",                   # -14.0
        "((2+3)*4)/5",                    # 4.0
        "-4 / 2",                         # -2.0
        "-5.5 + (1.5 * 2)",               # -2.5
        "2*(3 - (4/2))",                  # 2.0
        "2 / 0",                          # Raises ZeroDivisionError
        "2 + (3 *",                       # Raises ValueError (unbalanced parentheses)
        "2 + a",                          # Raises ValueError (invalid token)
        "--3",                            # 3.0
        "-(-3)",                          # 3.0
    ]
    for expr in expressions:
        try:
            result = calc.calculate(expr)
            print(f"{expr} = {result}")
        except Exception as e:
            print(f"{expr} -> Error: {e}")
