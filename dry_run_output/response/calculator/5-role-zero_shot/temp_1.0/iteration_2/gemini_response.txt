import re
from collections import deque
from typing import List, Union

class Calculator:
    """
    A high-quality, console-based arithmetic calculator.

    This class implements a calculator that can evaluate arithmetic expressions
    from a string. It supports addition, subtraction, multiplication, division,
    and parentheses, while respecting operator precedence.

    The implementation follows the principles of ISO/IEC 25010, focusing on
    correctness, performance, modularity, reliability, and maintainability.
    It uses the Shunting-yard algorithm to convert the infix expression to
    Reverse Polish Notation (RPN) and then evaluates the RPN expression.
    This approach avoids the use of `eval()`, enhancing safety.
    """

    # Operator properties: precedence and the actual function
    _OPERATORS = {
        '+': {'precedence': 1, 'func': lambda a, b: a + b},
        '-': {'precedence': 1, 'func': lambda a, b: a - b},
        '*': {'precedence': 2, 'func': lambda a, b: a * b},
        '/': {'precedence': 2, 'func': lambda a, b: a / b},
    }

    def calculate(self, expression: str) -> float:
        """
        Evaluates a mathematical expression provided as a string.

        Args:
            expression: The string containing the arithmetic expression.
                        e.g., "3 + 4 * (2 - 1)"

        Returns:
            The result of the evaluation as a float.

        Raises:
            ValueError: If the expression is invalid (e.g., mismatched parentheses,
                        invalid characters, malformed expression).
            ZeroDivisionError: If the expression involves division by zero.
        """
        self._validate_input(expression)
        tokens = self._tokenize(expression)
        rpn_queue = self._infix_to_rpn(tokens)
        result = self._evaluate_rpn(rpn_queue)
        return result

    def _validate_input(self, expression: str) -> None:
        """Performs initial validation on the input string."""
        if expression.count('(') != expression.count(')'):
            raise ValueError("Mismatched parentheses in expression.")
        
        # Check for any characters that are not allowed.
        allowed_chars = set("0123456789.+-*/() ")
        if not set(expression).issubset(allowed_chars):
            invalid_chars = set(expression) - allowed_chars
            raise ValueError(f"Invalid characters found: {', '.join(invalid_chars)}")

    def _tokenize(self, expression: str) -> List[str]:
        """
        Converts the input string into a list of tokens (numbers and operators).
        This process is also known as lexical analysis.

        It correctly handles multi-digit numbers, floating-point numbers, and
        unary minuses (e.g., '-5' or '(-5)').
        """
        # Add spaces around operators and parentheses to simplify splitting.
        # This regex looks for operators or parentheses and adds a space on both sides.
        expression = re.sub(r'([+\-*/()])', r' \1 ', expression)
        tokens = expression.split()

        # Handle unary minus: a '-' is unary if it's the first token, or
        # if it follows an operator or an opening parenthesis.
        processed_tokens = []
        for i, token in enumerate(tokens):
            if token == '-' and (i == 0 or tokens[i-1] in self._OPERATORS or tokens[i-1] == '('):
                # This is a unary minus. We prepend '0' to make it a binary operation.
                processed_tokens.append('0')
                processed_tokens.append('-')
            else:
                processed_tokens.append(token)
        return processed_tokens

    def _infix_to_rpn(self, tokens: List[str]) -> deque:
        """
        Converts a tokenized infix expression to Reverse Polish Notation (RPN)
        using the Shunting-yard algorithm.

        Args:
            tokens: A list of string tokens (numbers, operators, parentheses).

        Returns:
            A deque containing the expression in RPN order.

        Raises:
            ValueError: If the expression is malformed.
        """
        output_queue = deque()
        operator_stack = []

        for token in tokens:
            if token.replace('.', '', 1).isdigit():
                output_queue.append(float(token))
            elif token in self._OPERATORS:
                # While there's an operator on the stack with higher/equal precedence
                while (operator_stack and operator_stack[-1] in self._OPERATORS and
                       self._OPERATORS[operator_stack[-1]]['precedence'] >= self._OPERATORS[token]['precedence']):
                    output_queue.append(operator_stack.pop())
                operator_stack.append(token)
            elif token == '(':
                operator_stack.append(token)
            elif token == ')':
                while operator_stack and operator_stack[-1] != '(':
                    output_queue.append(operator_stack.pop())
                if not operator_stack or operator_stack.pop() != '(':
                    # This check is redundant if _validate_input is called,
                    # but provides robustness.
                    raise ValueError("Mismatched parentheses.")
            else:
                # This case should ideally not be reached if _validate_input and
                # tokenization are correct, but it's a safeguard.
                raise ValueError(f"Unknown token: {token}")

        while operator_stack:
            op = operator_stack.pop()
            if op == '(':
                raise ValueError("Mismatched parentheses.")
            output_queue.append(op)
            
        return output_queue

    def _evaluate_rpn(self, rpn_queue: deque) -> float:
        """
        Evaluates an expression in Reverse Polish Notation (RPN).

        Args:
            rpn_queue: A deque of numbers (float) and operators (str).

        Returns:
            The final result of the calculation.
        
        Raises:
            ValueError: If the expression is malformed (e.g., not enough operands).
            ZeroDivisionError: If division by zero is attempted.
        """
        evaluation_stack = []

        while rpn_queue:
            token = rpn_queue.popleft()
            
            if isinstance(token, float):
                evaluation_stack.append(token)
            elif token in self._OPERATORS:
                if len(evaluation_stack) < 2:
                    raise ValueError("Malformed expression: Not enough operands for an operator.")
                
                operand2 = evaluation_stack.pop()
                operand1 = evaluation_stack.pop()

                if token == '/' and operand2 == 0:
                    raise ZeroDivisionError("Division by zero is not allowed.")
                
                operation = self._OPERATORS[token]['func']
                result = operation(operand1, operand2)
                evaluation_stack.append(result)

        if len(evaluation_stack) != 1:
            raise ValueError("Malformed expression: The final stack should have one result.")
            
        return evaluation_stack[0]

        except Exception as e:
