"""
ascii_art.py

A console-based 2D ASCII Art application implementing shape drawing functions with input validation,
following ISO/IEC 25010 quality standards.

Author: [Your Name]
"""

import string


class AsciiArt:
    """
    Provides methods to draw various 2D ASCII shapes.

    Methods
    -------
    draw_square(width: int, symbol: str) -> str
        Draws a filled square of given width using the given symbol.

    draw_rectangle(width: int, height: int, symbol: str) -> str
        Draws a filled rectangle of given width and height using the given symbol.

    draw_parallelogram(width: int, height: int, symbol: str) -> str
        Draws a filled parallelogram (grows diagonally right) using the given symbol.

    draw_triangle(width: int, height: int, symbol: str) -> str
        Draws a right-angled triangle (top-left corner) using the given symbol.

    draw_pyramid(height: int, symbol: str) -> str
        Draws a symmetrical pyramid of given height using the given symbol.
    """

    def __init__(self):
        pass

    @staticmethod
    def _validate_symbol(symbol: str):
        """
        Validates that the symbol is a single, non-whitespace printable character.

        Raises
        ------
        ValueError
            If symbol is not a single printable, non-whitespace character.
        """
        if not isinstance(symbol, str):
            raise TypeError("Symbol must be a string.")
        if len(symbol) != 1:
            raise ValueError("Symbol must be a single character.")
        if symbol.isspace():
            raise ValueError("Symbol must not be a whitespace character.")
        if symbol not in string.printable.strip():
            raise ValueError("Symbol must be a printable character.")

    @staticmethod
    def _validate_positive_int(value: int, name: str):
        """
        Validates that the value is a positive integer.

        Raises
        ------
        ValueError
            If value is not a positive integer.
        """
        if not isinstance(value, int):
            raise TypeError(f"{name} must be an integer.")
        if value <= 0:
            raise ValueError(f"{name} must be a positive integer.")

    def draw_square(self, width: int, symbol: str) -> str:
        """
        Draws a filled square of given width using the symbol.

        Parameters
        ----------
        width : int
            The width (and height) of the square. Must be positive.
        symbol : str
            The symbol to use. Must be a single, non-whitespace printable character.

        Returns
        -------
        str
            The ASCII art string representing the square.
        """
        self._validate_positive_int(width, "Width")
        self._validate_symbol(symbol)

        line = symbol * width
        return '\n'.join([line] * width)

    def draw_rectangle(self, width: int, height: int, symbol: str) -> str:
        """
        Draws a filled rectangle of given width and height using the symbol.

        Parameters
        ----------
        width : int
            The width of the rectangle. Must be positive.
        height : int
            The height of the rectangle. Must be positive.
        symbol : str
            The symbol to use. Must be a single, non-whitespace printable character.

        Returns
        -------
        str
            The ASCII art string representing the rectangle.
        """
        self._validate_positive_int(width, "Width")
        self._validate_positive_int(height, "Height")
        self._validate_symbol(symbol)

        line = symbol * width
        return '\n'.join([line] * height)

    def draw_parallelogram(self, width: int, height: int, symbol: str) -> str:
        """
        Draws a filled parallelogram (grows diagonally right) using the symbol.

        Each line is shifted right by one space compared to the previous line.

        Parameters
        ----------
        width : int
            The width of the parallelogram. Must be positive.
        height : int
            The height of the parallelogram. Must be positive.
        symbol : str
            The symbol to use. Must be a single, non-whitespace printable character.

        Returns
        -------
        str
            The ASCII art string representing the parallelogram.
        """
        self._validate_positive_int(width, "Width")
        self._validate_positive_int(height, "Height")
        self._validate_symbol(symbol)

        lines = []
        for i in range(height):
            spaces = ' ' * i
            line = spaces + (symbol * width)
            lines.append(line)
        return '\n'.join(lines)

    def draw_triangle(self, width: int, height: int, symbol: str) -> str:
        """
        Draws a right-angled triangle (top-left corner) using the symbol.

        The triangle grows right and down, with each row's length proportional to its position.

        Parameters
        ----------
        width : int
            The width (base) of the triangle. Must be positive.
        height : int
            The height of the triangle. Must be positive.
        symbol : str
            The symbol to use. Must be a single, non-whitespace printable character.

        Returns
        -------
        str
            The ASCII art string representing the triangle.
        """
        self._validate_positive_int(width, "Width")
        self._validate_positive_int(height, "Height")
        self._validate_symbol(symbol)

        lines = []
        for i in range(height):
            # Linear interpolation: row_length varies from 1 to width as i goes from 0 to height-1
            if height == 1:
                row_length = width
            else:
                row_length = max(1, round(1 + (width - 1) * i / (height - 1)))
            lines.append(symbol * row_length)
        return '\n'.join(lines)

    def draw_pyramid(self, height: int, symbol: str) -> str:
        """
        Draws a symmetrical pyramid of given height using the symbol.

        The base width is (2 * height - 1).

        Parameters
        ----------
        height : int
            The height of the pyramid. Must be positive.
        symbol : str
            The symbol to use. Must be a single, non-whitespace printable character.

        Returns
        -------
        str
            The ASCII art string representing the pyramid.
        """
        self._validate_positive_int(height, "Height")
        self._validate_symbol(symbol)

        lines = []
        base_width = 2 * height - 1
        for i in range(height):
            num_symbols = 2 * i + 1
            spaces = ' ' * (height - i - 1)
            line = spaces + (symbol * num_symbols) + spaces
            lines.append(line)
        return '\n'.join(lines)

if __name__ == "__main__":
    art = AsciiArt()
    print("Square:\n", art.draw_square(4, '#'), sep='')
    print("\nRectangle:\n", art.draw_rectangle(6, 3, '*'), sep='')
    print("\nParallelogram:\n", art.draw_parallelogram(5, 4, '$'), sep='')
    print("\nTriangle:\n", art.draw_triangle(7, 5, '@'), sep='')
    print("\nPyramid:\n", art.draw_pyramid(5, '+'), sep='')
