"""
Provides a class for generating various 2D ASCII art shapes.

This module is designed to meet high-quality software standards, emphasizing
functional suitability, performance efficiency, maintainability, and reliability
as outlined by the ISO/IEC 25010 standard. It offers a simple, object-oriented
interface for creating console-based art.
"""

import sys

class AsciiArt:
    """
    A class to generate console-based 2D ASCII art shapes.

    This class encapsulates the logic for drawing various geometric shapes using
    ASCII characters. Each drawing method is a pure function that validates its
    inputs and returns a multi-line string representing the requested shape.
    This design ensures testability and reusability.

    Attributes:
        None
    """

    def _validate_input(self, symbol: str, **kwargs: int) -> None:
        """
        Validates the common inputs for drawing methods (internal helper).

        This method ensures that the provided symbol and dimensions meet the
        required criteria, contributing to the application's safety and
        fault tolerance.

        Args:
            symbol: The character to use for drawing.
            **kwargs: Keyword arguments for dimensions (e.g., width, height).

        Raises:
            TypeError: If the symbol is not a string or dimensions are not integers.
            ValueError: If the symbol is not a single, printable, non-whitespace
                        character, or if any dimension is not a positive integer.
        """
        # Validate symbol type and content
        if not isinstance(symbol, str):
            raise TypeError("Symbol must be a string.")
        if len(symbol) != 1:
            raise ValueError("Symbol must be a single character.")
        if not symbol.isprintable() or symbol.isspace():
            raise ValueError(
                "Symbol must be a printable, non-whitespace character."
            )

        # Validate dimension types and values
        for name, value in kwargs.items():
            if not isinstance(value, int):
                raise TypeError(f"{name.capitalize()} must be an integer.")
            if value <= 0:
                raise ValueError(f"{name.capitalize()} must be a positive integer.")

    def draw_square(self, width: int, symbol: str) -> str:
        """
        Draws a filled square.

        Args:
            width: The width and height of the square.
            symbol: The character to use for drawing the square.

        Returns:
            A multi-line string representing the square.
        """
        self._validate_input(symbol=symbol, width=width)
        line = symbol * width
        # Efficiently create the list of lines and join them
        return "\n".join([line] * width)

    def draw_rectangle(self, width: int, height: int, symbol: str) -> str:
        """
        Draws a filled rectangle.

        Args:
            width: The width of the rectangle.
            height: The height of the rectangle.
            symbol: The character to use for drawing the rectangle.

        Returns:
            A multi-line string representing the rectangle.
        """
        self._validate_input(symbol=symbol, width=width, height=height)
        line = symbol * width
        return "\n".join([line] * height)

    def draw_parallelogram(self, width: int, height: int, symbol: str) -> str:
        """
        Draws a filled parallelogram, skewed to the right.

        Each subsequent row is shifted one space to the right relative to the
        previous row.

        Args:
            width: The width of the parallelogram's top/bottom sides.
            height: The height of the parallelogram.
            symbol: The character to use for drawing.

        Returns:
            A multi-line string representing the parallelogram.
        """
        self._validate_input(symbol=symbol, width=width, height=height)
        # Using a list comprehension for concise and performant string generation
        lines = [" " * i + symbol * width for i in range(height)]
        return "\n".join(lines)

    def draw_triangle(self, width: int, height: int, symbol: str) -> str:
        """
        Draws a filled, right-angled triangle.

        The triangle's width grows linearly from top to bottom, fitting within
        the specified width and height.

        Args:
            width: The width of the triangle's base.
            height: The height of the triangle.
            symbol: The character to use for drawing.

        Returns:
            A multi-line string representing the right-angled triangle.
        """
        self._validate_input(symbol=symbol, width=width, height=height)

        # Handle the edge case of a single-line triangle to avoid division by zero
        if height == 1:
            return symbol * width

        lines = []
        # Calculate the growth rate (slope) of the triangle's side
        slope = (width - 1) / (height - 1)
        for i in range(height):
            # Determine the number of symbols for the current row using linear interpolation
            # We start with a width of 1 and grow towards `width`
            num_symbols = round(slope * i + 1)
            lines.append(symbol * num_symbols)
        return "\n".join(lines)

    def draw_pyramid(self, height: int, symbol: str) -> str:
        """
        Draws a filled, symmetrical pyramid.

        Args:
            height: The height of the pyramid in rows.
            symbol: The character to use for drawing.

        Returns:
            A multi-line string representing the pyramid.
        """
        self._validate_input(symbol=symbol, height=height)
        
        # The widest part of the pyramid determines the canvas width
        base_width = 2 * height - 1
        
        lines = []
        for i in range(height):
            num_symbols = 2 * i + 1
            # Center the symbols on each line by calculating the required padding
            padding = " " * ((base_width - num_symbols) // 2)
            lines.append(padding + symbol * num_symbols)
        return "\n".join(lines)
