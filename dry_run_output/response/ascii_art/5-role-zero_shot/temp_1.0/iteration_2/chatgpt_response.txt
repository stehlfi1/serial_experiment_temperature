```python
"""
AsciiArt - Console-based 2D ASCII Art Generator

This module defines the AsciiArt class which provides methods to generate
various ASCII shapes (square, rectangle, parallelogram, right-angled triangle,
and pyramid) for console output. The implementation emphasizes code quality,
input validation, and maintainability, in accordance with ISO/IEC 25010 standards.

Author: Senior Python Developer
"""

import string


class AsciiArt:
    """
    A class to generate 2D ASCII art shapes in string format.

    Methods
    -------
    draw_square(width: int, symbol: str) -> str
        Draws a filled square of given width with the specified symbol.

    draw_rectangle(width: int, height: int, symbol: str) -> str
        Draws a filled rectangle with the given width, height, and symbol.

    draw_parallelogram(width: int, height: int, symbol: str) -> str
        Draws a filled parallelogram shifted by one space each row.

    draw_triangle(width: int, height: int, symbol: str) -> str
        Draws a filled right-angled triangle (top-left right angle).

    draw_pyramid(height: int, symbol: str) -> str
        Draws a filled symmetrical pyramid of given height.
    """

    def __init__(self):
        pass  # No instance state required

    def draw_square(self, width: int, symbol: str) -> str:
        """
        Draws a square of the specified width using the given symbol.
        """
        self._validate_positive_int(width, "width")
        self._validate_symbol(symbol)
        lines = [symbol * width for _ in range(width)]
        return "\n".join(lines)

    def draw_rectangle(self, width: int, height: int, symbol: str) -> str:
        """
        Draws a rectangle of the specified width and height using the given symbol.
        """
        self._validate_positive_int(width, "width")
        self._validate_positive_int(height, "height")
        self._validate_symbol(symbol)
        lines = [symbol * width for _ in range(height)]
        return "\n".join(lines)

    def draw_parallelogram(self, width: int, height: int, symbol: str) -> str:
        """
        Draws a parallelogram with the given width and height.
        Each new row is shifted by one space to the right.
        """
        self._validate_positive_int(width, "width")
        self._validate_positive_int(height, "height")
        self._validate_symbol(symbol)
        lines = [(" " * i) + (symbol * width) for i in range(height)]
        return "\n".join(lines)

    def draw_triangle(self, width: int, height: int, symbol: str) -> str:
        """
        Draws a right-angled triangle (with the right angle at the top left).
        Fills the triangle with the specified symbol.

        The triangle is height rows tall, and the longest row is at the top.
        Each subsequent row has fewer symbols, with the final row containing
        at least one symbol.
        """
        self._validate_positive_int(width, "width")
        self._validate_positive_int(height, "height")
        self._validate_symbol(symbol)

        # The number of symbols per row decreases linearly from width->1.
        lines = []
        for row in range(height):
            # Linear decrement for each row
            symbols_in_row = max(
                1, width - (width - 1) * row // (height - 1) if height > 1 else width
            )
            lines.append(symbol * symbols_in_row)
        return "\n".join(lines)

    def draw_pyramid(self, height: int, symbol: str) -> str:
        """
        Draws a centered pyramid with the given height, using the symbol.
        The base is (2 * height - 1) symbols wide.
        """
        self._validate_positive_int(height, "height")
        self._validate_symbol(symbol)
        lines = []
        for row in range(height):
            num_symbols = 2 * row + 1
            num_spaces = height - row - 1
            line = (" " * num_spaces) + (symbol * num_symbols) + (" " * num_spaces)
            lines.append(line)
        return "\n".join(lines)

    @staticmethod
    def _validate_positive_int(value, name):
        """
        Validates that value is a positive integer.
        Raises ValueError if not.
        """
        if not isinstance(value, int):
            raise TypeError(f"{name} must be an integer.")
        if value <= 0:
            raise ValueError(f"{name} must be positive.")

    @staticmethod
    def _validate_symbol(symbol):
        """
        Validates that symbol is exactly one printable, non-whitespace character.
        Raises ValueError or TypeError if invalid.
        """
        if not isinstance(symbol, str):
            raise TypeError("Symbol must be a string.")
        if len(symbol) != 1:
            raise ValueError("Symbol must be a single character.")
        if symbol not in string.printable or symbol in string.whitespace:
            raise ValueError("Symbol must be a single, printable, non-whitespace character.")


if __name__ == "__main__":
    # Quick usage demonstration and testability sample
    art = AsciiArt()
    print("Square (width=4, symbol='#'):")
    print(art.draw_square(4, "#"))
    print("\nRectangle (width=6, height=3, symbol='*'):")
    print(art.draw_rectangle(6, 3, "*"))
    print("\nParallelogram (width=5, height=4, symbol='@'):")
    print(art.draw_parallelogram(5, 4, "@"))
    print("\nRight-angled Triangle (width=6, height=4, symbol='+'):")
    print(art.draw_triangle(6, 4, "+"))
    print("\nPyramid (height=5, symbol='X'):")
    print(art.draw_pyramid(5, "X"))
```

- **Correct**: Shapes are filled and aligned as specified.
