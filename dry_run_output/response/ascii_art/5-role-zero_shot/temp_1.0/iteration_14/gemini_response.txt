# -*- coding: utf-8 -*-
"""
ascii_art_generator.py

This module provides a class for generating various 2D ASCII art shapes.
It is designed with a focus on quality, maintainability, and reliability,
adhering to best practices in software development.
"""

import math
from typing import Any


class AsciiArt:
    """
    A class to generate various 2D ASCII art shapes as multi-line strings.

    This class encapsulates the logic for drawing different geometric shapes using
    ASCII characters. It emphasizes safety through robust input validation and
    promotes reusability and testability by returning string representations
    of the shapes.
    """

    def __init__(self) -> None:
        """Initializes the AsciiArt instance."""
        pass

    def _validate_inputs(self, symbol: str, **dimensions: Any) -> None:
        """
        A private helper method to validate common inputs for shape-drawing methods.

        This method centralizes input validation logic to ensure that shape
        dimensions are positive integers and the symbol is a single, non-whitespace
        character. This approach improves maintainability and fault tolerance.

        Args:
            symbol (str): The character to use for drawing.
            **dimensions (Any): Keyword arguments representing the dimensions
                                (e.g., width, height) of the shape.

        Raises:
            TypeError: If the symbol is not a string or a dimension is not an integer.
            ValueError: If the symbol is not a single character, is a whitespace,
                        or if any dimension is not a positive integer.
        """
        # Validate symbol type, length, and content
        if not isinstance(symbol, str):
            raise TypeError("Symbol must be a string.")
        if len(symbol) != 1:
            raise ValueError("Symbol must be a single character string.")
        if symbol.isspace():
            raise ValueError("Symbol cannot be a whitespace character.")

        # Validate all provided dimensions
        for name, value in dimensions.items():
            if not isinstance(value, int):
                raise TypeError(f"{name.capitalize()} must be an integer.")
            if value <= 0:
                raise ValueError(f"{name.capitalize()} must be a positive integer.")

    def draw_square(self, width: int, symbol: str) -> str:
        """
        Draws a filled square of a given width.

        This method leverages `draw_rectangle` for its implementation to promote
        code reuse and maintainability.

        Args:
            width (int): The width and height of the square. Must be positive.
            symbol (str): The single character symbol to use for drawing.

        Returns:
            str: A multi-line string representing the filled square.

        Raises:
            TypeError: If width is not an integer or symbol is not a string.
            ValueError: If width is not positive or symbol is invalid.
        """
        # A square is a rectangle with equal width and height.
        # Validation is handled by draw_rectangle.
        return self.draw_rectangle(width, width, symbol)

    def draw_rectangle(self, width: int, height: int, symbol: str) -> str:
        """
        Draws a filled rectangle of given width and height.

        This method uses an efficient list-based approach to build the final string,
        which offers better performance than repeated string concatenation.

        Args:
            width (int): The width of the rectangle. Must be positive.
            height (int): The height of the rectangle. Must be positive.
            symbol (str): The single character symbol to use for drawing.

        Returns:
            str: A multi-line string representing the filled rectangle.

        Raises:
            TypeError: If dimensions are not integers or symbol is not a string.
            ValueError: If dimensions are not positive or symbol is invalid.
        """
        self._validate_inputs(symbol, width=width, height=height)
        
        line = symbol * width
        # Create a list of lines and join them, which is more efficient
        # than concatenating strings in a loop.
        lines = [line] * height
        return "\n".join(lines)

    def draw_parallelogram(self, width: int, height: int, symbol:str) -> str:
        """
        Draws a filled parallelogram that slants to the right.

        Each subsequent row is indented by one additional space to create the
        slanting effect. The shape is constructed using a list comprehension
        for conciseness and efficiency.

        Args:
            width (int): The width of the parallelogram base. Must be positive.
            height (int): The height of the parallelogram. Must be positive.
            symbol (str): The single character symbol to use for drawing.

        Returns:
            str: A multi-line string representing the filled parallelogram.
        
        Raises:
            TypeError: If dimensions are not integers or symbol is not a string.
            ValueError: If dimensions are not positive or symbol is invalid.
        """
        self._validate_inputs(symbol, width=width, height=height)

        lines = [
            ' ' * i + symbol * width
            for i in range(height)
        ]
        return "\n".join(lines)

    def draw_triangle(self, width: int, height: int, symbol: str) -> str:
        """
        Draws a filled, right-angled triangle.

        The triangle grows from the top-left corner. The width of each row is
        calculated proportionally to its position to ensure the base of the
        triangle matches the specified width.

        Args:
            width (int): The width of the triangle's base. Must be positive.
            height (int): The height of the triangle. Must be positive.
            symbol (str): The single character symbol to use for drawing.

        Returns:
            str: A multi-line string representing the right-angled triangle.

        Raises:
            TypeError: If dimensions are not integers or symbol is not a string.
            ValueError: If dimensions are not positive or symbol is invalid.
        """
        self._validate_inputs(symbol, width=width, height=height)

        lines = []
        for i in range(1, height + 1):
            # Calculate the number of symbols for the current row based on its
            # proportion to the total height and width. Using math.ceil ensures
            # the final row has the correct width.
            num_symbols = math.ceil(i * (width / height))
            lines.append(symbol * num_symbols)
        return "\n".join(lines)

    def draw_pyramid(self, height: int, symbol: str) -> str:
        """
        Draws a symmetrical, filled pyramid.

        The pyramid is centered by calculating the necessary padding for each row.
        Each row `i` has `2*i - 1` symbols.

        Args:
            height (int): The height of the pyramid. Must be positive.
            symbol (str): The single character symbol to use for drawing.

        Returns:
            str: A multi-line string representing the pyramid.
            
        Raises:
            TypeError: If height is not an integer or symbol is not a string.
            ValueError: If height is not a positive integer or symbol is invalid.
        """
        self._validate_inputs(symbol, height=height)

        lines = [
            ' ' * (height - i) + symbol * (2 * i - 1)
            for i in range(1, height + 1)
        ]
        return "\n".join(lines)
