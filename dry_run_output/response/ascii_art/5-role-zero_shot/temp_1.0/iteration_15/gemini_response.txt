# -*- coding: utf-8 -*-
"""
A module for generating 2D ASCII art shapes.

This module provides the AsciiArt class, which offers a robust interface for
drawing various geometric shapes using ASCII characters. The implementation
adheres to high-quality code standards, focusing on correctness, efficiency,
safety, and maintainability, in line with ISO/IEC 25010 principles.
"""

import sys
from typing import List


class AsciiArt:
    """
    A class for generating console-based 2D ASCII art shapes.

    This class provides a collection of methods to draw various geometric shapes
    filled with a user-specified symbol. It is designed with a focus on
    modularity, reusability, and safety, ensuring all inputs are validated
    before processing.

    Attributes:
        None
    """

    def _validate_dimensions(self, *dims: int) -> None:
        """
        Private helper to validate shape dimensions.

        Ensures dimensions are integers and are positive.

        Args:
            *dims: A variable number of integer dimensions to validate.

        Raises:
            TypeError: If any dimension is not an integer.
            ValueError: If any dimension is not a positive number (> 0).
        """
        for i, dim in enumerate(dims):
            if not isinstance(dim, int):
                raise TypeError(f"Dimension argument {i+1} must be an integer, not {type(dim).__name__}.")
            if dim <= 0:
                raise ValueError(f"Dimension argument {i+1} must be positive, not {dim}.")

    def _validate_symbol(self, symbol: str) -> None:
        """
        Private helper to validate the drawing symbol.

        Ensures the symbol is a single, printable, non-whitespace character.

        Args:
            symbol: The character to be used for drawing.

        Raises:
            TypeError: If the symbol is not a string.
            ValueError: If the symbol is not a single character, is whitespace,
                        or is not printable.
        """
        if not isinstance(symbol, str):
            raise TypeError(f"Symbol must be a string, not {type(symbol).__name__}.")
        if len(symbol) != 1:
            raise ValueError(f"Symbol must be a single character, not '{symbol}'.")
        if symbol.isspace():
            raise ValueError("Symbol cannot be a whitespace character.")
        if not symbol.isprintable():
            raise ValueError(f"Symbol must be a printable character, not '{repr(symbol)}'.")

    def draw_square(self, width: int, symbol: str) -> str:
        """
        Draws a filled square of a given width.

        Args:
            width: The width and height of the square.
            symbol: The character used to draw the square.

        Returns:
            A multi-line string representing the ASCII square.

        Raises:
            TypeError: If width is not an integer or symbol is not a string.
            ValueError: If width is not positive or symbol is invalid.
        """
        self._validate_dimensions(width)
        self._validate_symbol(symbol)
        
        # For a square, height is the same as the width.
        return self.draw_rectangle(width, width, symbol)

    def draw_rectangle(self, width: int, height: int, symbol: str) -> str:
        """
        Draws a filled rectangle of given width and height.

        Args:
            width: The width of the rectangle.
            height: The height of the rectangle.
            symbol: The character used to draw the rectangle.

        Returns:
            A multi-line string representing the ASCII rectangle.
            
        Raises:
            TypeError: If dimensions aren't integers or symbol isn't a string.
            ValueError: If dimensions aren't positive or symbol is invalid.
        """
        self._validate_dimensions(width, height)
        self._validate_symbol(symbol)
        
        line = symbol * width
        # Use list comprehension and join for efficient string building,
        # avoiding slow concatenation in a loop.
        lines = [line for _ in range(height)]
        return "\n".join(lines)

    def draw_parallelogram(self, width: int, height: int, symbol: str) -> str:
        """
        Draws a filled parallelogram.

        The shape grows diagonally to the right, with each subsequent row
        being indented by one additional space.

        Args:
            width: The width of the parallelogram's top/bottom sides.
            height: The height of the parallelogram.
            symbol: The character used to draw the shape.

        Returns:
            A multi-line string representing the ASCII parallelogram.

        Raises:
            TypeError: If dimensions aren't integers or symbol isn't a string.
            ValueError: If dimensions aren't positive or symbol is invalid.
        """
        self._validate_dimensions(width, height)
        self._validate_symbol(symbol)
        
        lines: List[str] = []
        base_line = symbol * width
        for i in range(height):
            # Each row is indented by 'i' spaces.
            indent = " " * i
            lines.append(f"{indent}{base_line}")
        return "\n".join(lines)

    def draw_triangle(self, width: int, height: int, symbol: str) -> str:
        """
        Draws a filled, right-angled triangle.

        The triangle's width scales proportionally with its height, starting
        from the top-left corner. The final row will have the specified width.

        Args:
            width: The width of the triangle's base.
            height: The height of the triangle.
            symbol: The character used to draw the shape.

        Returns:
            A multi-line string representing the ASCII triangle.
        
        Raises:
            TypeError: If dimensions aren't integers or symbol isn't a string.
            ValueError: If dimensions aren't positive or symbol is invalid.
        """
        self._validate_dimensions(width, height)
        self._validate_symbol(symbol)
        
        lines: List[str] = []
        for i in range(1, height + 1):
            # Calculate the width for the current row based on a linear slope.
            # Using max(1, ...) ensures at least one symbol is printed per row.
            current_width = max(1, int(i / height * width))
            lines.append(symbol * current_width)
        return "\n".join(lines)

    def draw_pyramid(self, height: int, symbol: str) -> str:
        """
        Draws a symmetrical, filled pyramid.

        Args:
            height: The height of the pyramid in rows.
            symbol: The character used to draw the shape.

        Returns:
            A multi-line string representing the ASCII pyramid.

        Raises:
            TypeError: If height is not an integer or symbol is not a string.
            ValueError: If height is not positive or symbol is invalid.
        """
        self._validate_dimensions(height)
        self._validate_symbol(symbol)
        
        lines: List[str] = []
        # The width of the base is (2 * height - 1)
        # This can be derived from the number of symbols in the last row.
        for i in range(height):
            # Number of symbols in the current row (0-indexed)
            num_symbols = 2 * i + 1
            # Number of spaces for padding on the left to center the row
            padding = " " * (height - 1 - i)
            row_content = symbol * num_symbols
            lines.append(f"{padding}{row_content}")
        return "\n".join(lines)
