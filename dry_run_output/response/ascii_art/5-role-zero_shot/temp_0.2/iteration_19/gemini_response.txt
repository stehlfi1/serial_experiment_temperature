"""
A module for generating 2D ASCII art shapes.

This module provides the AsciiArt class, which implements methods for drawing
various geometric shapes using ASCII characters. The implementation prioritizes
code quality, adhering to principles of the ISO/IEC 25010 standard, including
modularity, reliability, and efficiency.

Classes:
    AsciiArt: A class to generate various ASCII art shapes.
"""

from typing import Dict, Any

class AsciiArt:
    """
    A class to generate various filled, 2D ASCII art shapes.

    This class provides a suite of methods to draw common geometric shapes
    like squares, rectangles, and pyramids as multi-line strings. It is designed
    for reliability and ease of use, with built-in validation for all inputs.

    Methods:
        draw_square(width, symbol): Draws a square.
        draw_rectangle(width, height, symbol): Draws a rectangle.
        draw_parallelogram(width, height, symbol): Draws a parallelogram.
        draw_triangle(width, height, symbol): Draws a right-angled triangle.
        draw_pyramid(height, symbol): Draws a symmetrical pyramid.
    """

    def _validate_inputs(self, symbol: str, **dimensions: int) -> None:
        """
        Validates the symbol and dimension inputs for drawing methods.

        This private helper method ensures that the provided symbol is a single,
        non-whitespace character and that all dimensions (e.g., width, height)
        are positive integers. It centralizes input validation to ensure
        consistency and safety across all drawing methods.

        Args:
            symbol (str): The character to use for drawing.
            **dimensions (int): A dictionary of dimension names and their integer values.

        Raises:
            TypeError: If the symbol is not a string or a dimension is not an integer.
            ValueError: If the symbol is not a single character, is whitespace,
                        or if any dimension is not a positive integer.
        """
        if not isinstance(symbol, str):
            raise TypeError("Symbol must be a string.")
        if len(symbol) != 1:
            raise ValueError("Symbol must be a single character.")
        if symbol.isspace():
            raise ValueError("Symbol cannot be a whitespace character.")

        for name, value in dimensions.items():
            if not isinstance(value, int):
                raise TypeError(f"{name.capitalize()} must be an integer.")
            if value <= 0:
                raise ValueError(f"{name.capitalize()} must be a positive integer.")

    def draw_square(self, width: int, symbol: str) -> str:
        """
        Draws a filled square of a given width.

        Args:
            width (int): The width and height of the square. Must be a positive integer.
            symbol (str): The single, non-whitespace character to fill the square with.

        Returns:
            str: A multi-line string representing the ASCII square.

        Raises:
            TypeError: If width is not an integer or symbol is not a string.
            ValueError: If width is not positive or symbol is invalid.
        """
        # A square is a rectangle with equal width and height.
        # This delegation improves maintainability (DRY principle).
        return self.draw_rectangle(width=width, height=width, symbol=symbol)

    def draw_rectangle(self, width: int, height: int, symbol:str) -> str:
        """
        Draws a filled rectangle of given width and height.

        Args:
            width (int): The width of the rectangle. Must be a positive integer.
            height (int): The height of the rectangle. Must be a positive integer.
            symbol (str): The single, non-whitespace character to fill the rectangle with.

        Returns:
            str: A multi-line string representing the ASCII rectangle.

        Raises:
            TypeError: If width/height are not integers or symbol is not a string.
            ValueError: If width/height are not positive or symbol is invalid.
        """
        self._validate_inputs(symbol, width=width, height=height)
        row = symbol * width
        # Using a list comprehension and join is more performant than string
        # concatenation in a loop for building multi-line strings.
        rows = [row] * height
        return "\n".join(rows)

    def draw_parallelogram(self, width: int, height: int, symbol: str) -> str:
        """
        Draws a filled parallelogram, skewed to the right.

        Each subsequent row is shifted one space to the right relative to the one above it.

        Args:
            width (int): The width of the parallelogram. Must be a positive integer.
            height (int): The height of the parallelogram. Must be a positive integer.
            symbol (str): The single, non-whitespace character to fill the shape with.

        Returns:
            str: A multi-line string representing the ASCII parallelogram.

        Raises:
            TypeError: If width/height are not integers or symbol is not a string.
            ValueError: If width/height are not positive or symbol is invalid.
        """
        self._validate_inputs(symbol, width=width, height=height)
        rows = []
        for i in range(height):
            padding = " " * i
            content = symbol * width
            rows.append(f"{padding}{content}")
        return "\n".join(rows)

    def draw_triangle(self, width: int, height: int, symbol: str) -> str:
        """
        Draws a filled, right-angled triangle.

        The triangle grows from a single point at the top-left to a base
        of the specified width, scaled proportionally to the height.

        Args:
            width (int): The width of the triangle's base. Must be a positive integer.
            height (int): The height of the triangle. Must be a positive integer.
            symbol (str): The single, non-whitespace character to fill the triangle with.

        Returns:
            str: A multi-line string representing the ASCII triangle.

        Raises:
            TypeError: If width/height are not integers or symbol is not a string.
            ValueError: If width/height are not positive or symbol is invalid.
        """
        self._validate_inputs(symbol, width=width, height=height)
        rows = []
        for i in range(1, height + 1):
            # Calculate the number of symbols for the current row based on its
            # proportion to the total height and width. This ensures the triangle
            # scales correctly for any width/height combination.
            # max(1, ...) ensures that even for very thin triangles, each row has at least one symbol.
            current_width = max(1, round(i * width / height))
            rows.append(symbol * current_width)
        return "\n".join(rows)

    def draw_pyramid(self, height: int, symbol: str) -> str:
        """
        Draws a filled, symmetrical pyramid.

        Args:
            height (int): The height of the pyramid. Must be a positive integer.
            symbol (str): The single, non-whitespace character to fill the pyramid with.

        Returns:
            str: A multi-line string representing the ASCII pyramid.

        Raises:
            TypeError: If height is not an integer or symbol is not a string.
            ValueError: If height is not positive or symbol is invalid.
        """
        self._validate_inputs(symbol, height=height)
        rows = []
        # The base of the pyramid has a width of (2 * height - 1).
        # This is used to calculate the necessary padding for centering.
        for i in range(1, height + 1):
            num_symbols = 2 * i - 1
            padding = " " * (height - i)
            content = symbol * num_symbols
            rows.append(f"{padding}{content}")
        return "\n".join(rows)
