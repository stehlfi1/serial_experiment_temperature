"""
ascii_art.py

A console-based ASCII Art application for drawing basic 2D shapes using OOP.
Fulfills maintainability, correctness, reliability, and usability requirements

Author: Senior Software Developer

from typing import Optional

class AsciiArt:
    """
    Provides methods to generate ASCII art representations of geometric shapes.
    """

    @staticmethod
    def _validate_symbol(symbol: str) -> None:
        """
        Validates the symbol to ensure it is a single, non-whitespace, printable character.

        Raises:
            ValueError: If the symbol is empty, more than one character, or is a whitespace/non-printable.
        """
        if not isinstance(symbol, str):
            raise TypeError("Symbol must be of type 'str'.")
        if len(symbol) != 1:
            raise ValueError("Symbol must be a single character.")
        if symbol.isspace():
            raise ValueError("Symbol must not be a whitespace character.")
        if not symbol.isprintable():
            raise ValueError("Symbol must be a printable character.")

    @staticmethod
    def _validate_positive_int(value: int, name: Optional[str] = None) -> None:
        """
        Validates that a value is a positive integer.

        Args:
            value (int): The value to validate.
            name (str, optional): The parameter name, for error messaging.

        Raises:
            TypeError: If the value is not an integer.
            ValueError: If the value is not positive.
        """
        if not isinstance(value, int):
            raise TypeError(f"{name or 'Value'} must be of type 'int'.")
        if value <= 0:
            raise ValueError(f"{name or 'Value'} must be a positive integer.")

    def draw_square(self, width: int, symbol: str) -> str:
        """
        Draws a filled square ASCII art.

        Args:
            width (int): The width (and height) of the square (must be > 0).
            symbol (str): The symbol used for drawing (single printable non-whitespace character).

        Returns:
            str: A string containing the ASCII art of the square.
        """
        self._validate_positive_int(width, "Width")
        self._validate_symbol(symbol)
        row = symbol * width
        # Using join for efficiency
        return '\n'.join([row for _ in range(width)])

    def draw_rectangle(self, width: int, height: int, symbol: str) -> str:
        """
        Draws a filled rectangle ASCII art.

        Args:
            width (int): The width of the rectangle (must be > 0).
            height (int): The height of the rectangle (must be > 0).
            symbol (str): The symbol used for drawing (single printable non-whitespace character).

        Returns:
            str: A string containing the ASCII art of the rectangle.
        """
        self._validate_positive_int(width, "Width")
        self._validate_positive_int(height, "Height")
        self._validate_symbol(symbol)
        row = symbol * width
        return '\n'.join([row for _ in range(height)])

    def draw_parallelogram(self, width: int, height: int, symbol: str) -> str:
        """
        Draws a filled parallelogram ASCII art (each row is shifted right by one space).

        Args:
            width (int): The width of the parallelogram (must be > 0).
            height (int): The height of the parallelogram (must be > 0).
            symbol (str): The symbol used for drawing (single printable non-whitespace character).

        Returns:
            str: A string containing the ASCII art of the parallelogram.
        """
        self._validate_positive_int(width, "Width")
        self._validate_positive_int(height, "Height")
        self._validate_symbol(symbol)
        lines = []
        for row in range(height):
            # Each row shifted right by its index
            lines.append(' ' * row + symbol * width)
        return '\n'.join(lines)

    def draw_triangle(self, width: int, height: int, symbol: str) -> str:
        """
        Draws a filled right-angled triangle ASCII art, growing from the top-left.

        Args:
            width (int): The width of the triangle's base (must be > 0).
            height (int): The height of the triangle (must be > 0).
            symbol (str): The symbol used for drawing (single printable non-whitespace character).

        Returns:
            str: A string containing the ASCII art of the triangle.
        """
        self._validate_positive_int(width, "Width")
        self._validate_positive_int(height, "Height")
        self._validate_symbol(symbol)

        lines = []
        for i in range(height):
            # Compute number of symbols for current row:
            # Linear interpolation between 1 (top) and width (bottom)
            num_symbols = max(1, (width * (i + 1)) // height)
            lines.append(symbol * num_symbols)
        return '\n'.join(lines)

    def draw_pyramid(self, height: int, symbol: str) -> str:
        """
        Draws a filled symmetrical pyramid ASCII art.

        Args:
            height (int): The number of rows for the pyramid (must be > 0).
            symbol (str): The symbol used for drawing (single printable non-whitespace character).

        Returns:
            str: A string containing the ASCII art of the pyramid.
        """
        self._validate_positive_int(height, "Height")
        self._validate_symbol(symbol)

        lines = []
        for i in range(height):
            num_symbols = 2 * i + 1
            spaces = height - i - 1
            lines.append(' ' * spaces + symbol * num_symbols)
        return '\n'.join(lines)

if __name__ == "__main__":
    art = AsciiArt()

    print("Square:")
    print(art.draw_square(4, "*"))
    print("\nRectangle:")
    print(art.draw_rectangle(6, 3, "#"))
    print("\nParallelogram:")
    print(art.draw_parallelogram(5, 4, "@"))
    print("\nTriangle:")
    print(art.draw_triangle(6, 4, "+"))
    print("\nPyramid:")
    print(art.draw_pyramid(5, "$"))
