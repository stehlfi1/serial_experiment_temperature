```python
import math
from typing import Dict, Any

class AsciiArt:
    """
    A class to generate various 2D ASCII art shapes.

    This class provides methods to draw simple geometric shapes using
    ASCII characters, returning them as multi-line strings. It adheres to
    ISO/IEC 25010 quality standards by ensuring correctness, performance,
    modularity, and safety through robust input validation.
    """

    @staticmethod
    def _validate_inputs(symbol: str, **kwargs: Any) -> None:
        """
        Validates the common inputs for drawing methods.

        :param symbol: The character to validate.
        :param kwargs: A dictionary of named dimensions (e.g., width, height) to validate.
        :raises TypeError: If a dimension is not an integer.
        :raises ValueError: If a symbol is invalid or a dimension is not positive.
        """
        # 1. Validate symbol
        if not isinstance(symbol, str) or len(symbol) != 1:
            raise ValueError("Symbol must be a single character string.")
        if symbol.isspace():
            raise ValueError("Symbol cannot be a whitespace character.")

        # 2. Validate dimensions (width, height, etc.)
        for name, value in kwargs.items():
            if not isinstance(value, int):
                raise TypeError(f"{name.capitalize()} must be an integer.")
            if value <= 0:
                raise ValueError(f"{name.capitalize()} must be a positive integer.")

    def draw_square(self, width: int, symbol: str) -> str:
        """
        Draws a filled square of a given width.

        :param width: The width and height of the square. Must be a positive integer.
        :param symbol: The character used for drawing. Must be a single,
                       non-whitespace character.
        :return: A multi-line string representing the square.
        :raises TypeError: If width is not an integer.
        :raises ValueError: If width is not positive, or if the symbol is invalid.
        """
        self._validate_inputs(symbol, width=width)
        
        row = symbol * width
        rows = [row] * width
        return "\n".join(rows)

    def draw_rectangle(self, width: int, height: int, symbol: str) -> str:
        """
        Draws a filled rectangle of a given width and height.

        :param width: The width of the rectangle. Must be a positive integer.
        :param height: The height of the rectangle. Must be a positive integer.
        :param symbol: The character used for drawing. Must be a single,
                       non-whitespace character.
        :return: A multi-line string representing the rectangle.
        :raises TypeError: If width or height are not integers.
        :raises ValueError: If width or height are not positive, or if the symbol is invalid.
        """
        self._validate_inputs(symbol, width=width, height=height)
        
        row = symbol * width
        rows = [row] * height
        return "\n".join(rows)

    def draw_parallelogram(self, width: int, height: int, symbol: str) -> str:
        """
        Draws a filled parallelogram that leans to the right.

        Each subsequent row is shifted one space to the right.

        :param width: The width of the parallelogram's top/bottom side. Must be a positive integer.
        :param height: The height of the parallelogram. Must be a positive integer.
        :param symbol: The character used for drawing. Must be a single,
                       non-whitespace character.
        :return: A multi-line string representing the parallelogram.
        :raises TypeError: If width or height are not integers.
        :raises ValueError: If width or height are not positive, or if the symbol is invalid.
        """
        self._validate_inputs(symbol, width=width, height=height)
        
        rows = [" " * i + symbol * width for i in range(height)]
        return "\n".join(rows)

    def draw_triangle(self, width: int, height: int, symbol: str) -> str:
        """
        Draws a filled, right-angled triangle.

        The triangle grows from the top-left corner to fit within the specified
        width and height. The number of symbols per row is scaled to ensure
        the base of the triangle has the specified width.

        :param width: The width of the triangle's base. Must be a positive integer.
        :param height: The height of the triangle. Must be a positive integer.
        :param symbol: The character used for drawing. Must be a single,
                       non-whitespace character.
        :return: A multi-line string representing the right-angled triangle.
        :raises TypeError: If width or height are not integers.
        :raises ValueError: If width or height are not positive, or if the symbol is invalid.
        """
        self._validate_inputs(symbol, width=width, height=height)
        
        rows = []
        for i in range(height):
            # Scale the number of symbols for the current row to fit the bounding box.
            # We use ceiling to ensure the last row has the correct width and
            # the first row has at least one symbol.
            num_symbols = math.ceil((i + 1) * width / height)
            rows.append(symbol * num_symbols)
            
        return "\n".join(rows)

    def draw_pyramid(self, height: int, symbol: str) -> str:
        """
        Draws a symmetrical, filled pyramid.

        :param height: The height of the pyramid in rows. Must be a positive integer.
        :param symbol: The character used for drawing. Must be a single,
                       non-whitespace character.
        :return: A multi-line string representing the pyramid.
        :raises TypeError: If height is not an integer.
        :raises ValueError: If height is not positive, or if the symbol is invalid.
        """
        self._validate_inputs(symbol, height=height)
        
        rows = []
        for i in range(height):
            num_spaces = height - 1 - i
            num_symbols = 2 * i + 1
            rows.append(" " * num_spaces + symbol * num_symbols)
            
        return "\n".join(rows)
```
