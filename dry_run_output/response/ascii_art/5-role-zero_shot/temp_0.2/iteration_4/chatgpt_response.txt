```python
"""
ASCII Art Generator - Console-based 2D ASCII Art Drawing Application

This module defines an AsciiArt class for generating filled ASCII shapes
in the console, adhering to ISO/IEC 25010 software quality standards:
- Modularity: Each drawing operation is encapsulated in a method.
- Correctness, Safety: Full parameter validation, well-documented.
- Testability: All outputs are pure functions (no I/O, easy to test).
- Maintainability: Readable structure and consistent style.
"""

from typing import Any

class AsciiArt:
    """
    ASCII Art class for generating filled shapes as multi-line strings.
    Provides robust parameter validation according to quality standards.
    """

    def _validate_symbol(self, symbol: str) -> None:
        """
        Validates the printable symbol.
        Rules:
            - Must be a single, non-whitespace, printable character.
        Raises:
            ValueError: If symbol doesn't meet constraints.
        """
        if not isinstance(symbol, str):
            raise TypeError("Symbol must be a string.")
        if len(symbol) != 1:
            raise ValueError("Symbol must be a single character.")
        if symbol.isspace():
            raise ValueError("Symbol cannot be a whitespace character.")
        if not symbol.isprintable():
            raise ValueError("Symbol must be printable.")

    def _validate_positive_int(self, value: Any, name: str) -> int:
        """
        Confirms if value is a positive integer.
        Args:
            value: Any - value to check
            name: str - parameter name for error messages
        
        Returns:
            int - Validated, cast value.
        
        Raises:
            ValueError or TypeError
        """
        if not isinstance(value, int):
            raise TypeError(f"{name} must be an integer.")
        if value <= 0:
            raise ValueError(f"{name} must be positive.")
        return value

    def draw_square(self, width: int, symbol: str) -> str:
        """
        Draws a filled square as a multi-line string.

        Args:
            width (int): Side length
            symbol (str): Fill character

        Returns:
            str: ASCII art string
        """
        width = self._validate_positive_int(width, "width")
        self._validate_symbol(symbol)

        row = symbol * width
        lines = [row for _ in range(width)]
        return "\n".join(lines)

    def draw_rectangle(self, width: int, height: int, symbol: str) -> str:
        """
        Draws a filled rectangle as a multi-line string.

        Args:
            width (int): Rectangle width
            height (int): Rectangle height
            symbol (str): Fill character

        Returns:
            str: ASCII art string
        """
        width = self._validate_positive_int(width, "width")
        height = self._validate_positive_int(height, "height")
        self._validate_symbol(symbol)

        row = symbol * width
        lines = [row for _ in range(height)]
        return "\n".join(lines)

    def draw_parallelogram(self, width: int, height: int, symbol: str) -> str:
        """
        Draws a filled parallelogram (shift right each row) as a multi-line string.

        Args:
            width (int): Number of symbols per row
            height (int): Number of rows
            symbol (str): Fill character

        Returns:
            str: ASCII art string
        """
        width = self._validate_positive_int(width, "width")
        height = self._validate_positive_int(height, "height")
        self._validate_symbol(symbol)

        lines = []
        for i in range(height):
            spaces = ' ' * i
            row = spaces + (symbol * width)
            lines.append(row)
        return "\n".join(lines)

    def draw_triangle(self, width: int, height: int, symbol: str) -> str:
        """
        Draws a filled right-angled triangle (top-left) as a multi-line string.

        Args:
            width (int): Maximum width at base
            height (int): Number of rows
            symbol (str): Fill character

        Returns:
            str: ASCII art string
        """
        width = self._validate_positive_int(width, "width")
        height = self._validate_positive_int(height, "height")
        self._validate_symbol(symbol)

        lines = []
        for row in range(height):
            # For "grows to right from top-left", each row increases in length
            count = max(1, int((row + 1) * width / height))
            lines.append(symbol * count)
        return "\n".join(lines)

    def draw_pyramid(self, height: int, symbol: str) -> str:
        """
        Draws a symmetrical filled pyramid as a multi-line string.

        Args:
            height (int): Number of rows
            symbol (str): Fill character

        Returns:
            str: ASCII art string
        """
        height = self._validate_positive_int(height, "height")
        self._validate_symbol(symbol)

        lines = []
        for row in range(height):
            spaces = ' ' * (height - row - 1)
            symbols = symbol * (2 * row + 1)
            lines.append(f"{spaces}{symbols}{spaces}")
        return "\n".join(lines)

# For demonstration and testing (Not part of the module's core functionality)
if __name__ == "__main__":
    art = AsciiArt()
    print("Square:\n", art.draw_square(4, "#"))
    print("\nRectangle:\n", art.draw_rectangle(6, 3, "*"))
    print("\nParallelogram:\n", art.draw_parallelogram(5, 4, "+"))
    print("\nTriangle:\n", art.draw_triangle(7, 5, "@"))
    print("\nPyramid:\n", art.draw_pyramid(4, "$"))
```
