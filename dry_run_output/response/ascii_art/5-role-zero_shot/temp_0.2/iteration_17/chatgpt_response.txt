```python
"""
ascii_art.py

Console-based 2D ASCII ART application.

Author: [Your Name]
Adheres to ISO/IEC 25010 quality model: Functionality, Reliability, Usability,
Efficiency, Maintainability, Portability, and Security.
"""

import string

class AsciiArt:
    """
    Class to generate ASCII art for basic 2D shapes.
    
    Exported methods:
        - draw_square
        - draw_rectangle
        - draw_parallelogram
        - draw_triangle
        - draw_pyramid
    """

    def __init__(self):
        pass

    @staticmethod
    def _validate_dim(value: int, name: str):
        """
        Validates that a dimension is a positive integer (>=1).
        """
        if not isinstance(value, int):
            raise TypeError(f"{name} must be an integer.")
        if value < 1:
            raise ValueError(f"{name} must be >= 1.")

    @staticmethod
    def _validate_symbol(symbol: str):
        """
        Validates that the symbol is a single, printable, non-whitespace character.
        """
        if not isinstance(symbol, str):
            raise TypeError("Symbol must be a string.")
        if len(symbol) != 1:
            raise ValueError("Symbol must be a single character.")
        if symbol.isspace():
            raise ValueError("Symbol cannot be a whitespace character.")
        if symbol not in string.printable or symbol in string.whitespace:
            raise ValueError("Symbol must be a printable, non-whitespace character.")

    def draw_square(self, width: int, symbol: str) -> str:
        """
        Draws a filled square of given width using the specified symbol.
        
        Returns:
            str: Multi-line string representing the square.
        """
        self._validate_dim(width, "Width")
        self._validate_symbol(symbol)
        square = '\n'.join([symbol * width for _ in range(width)])
        return square

    def draw_rectangle(self, width: int, height: int, symbol: str) -> str:
        """
        Draws a filled rectangle of width and height using symbol.

        Returns:
            str: Multi-line string representing the rectangle.
        """
        self._validate_dim(width, "Width")
        self._validate_dim(height, "Height")
        self._validate_symbol(symbol)
        rectangle = '\n'.join([symbol * width for _ in range(height)])
        return rectangle

    def draw_parallelogram(self, width: int, height: int, symbol: str) -> str:
        """
        Draws a filled parallelogram of given width and height, growing diagonally to the right.
        Each row is shifted by one space to the right compared to the one above.
        
        Returns:
            str: Multi-line string representing the parallelogram.
        """
        self._validate_dim(width, "Width")
        self._validate_dim(height, "Height")
        self._validate_symbol(symbol)
        lines = []
        for row in range(height):
            line = ' ' * row + symbol * width
            lines.append(line)
        return '\n'.join(lines)

    def draw_triangle(self, width: int, height: int, symbol: str) -> str:
        """
        Draws a filled right-angled triangle (top-left right angle), growing rightward and downward.
        If width and height are unequal, height controls the rows, width the maximum columns.

        Returns:
            str: Multi-line string representing the right-angled triangle.
        """
        self._validate_dim(width, "Width")
        self._validate_dim(height, "Height")
        self._validate_symbol(symbol)
        lines = []
        for row in range(height):
            # Determine the number of symbols for the current row.
            current_width = min(width, row + 1)
            line = symbol * current_width
            lines.append(line)
        return '\n'.join(lines)

    def draw_pyramid(self, height: int, symbol: str) -> str:
        """
        Draws a symmetrical filled pyramid of a given height using symbol.
        Centrally aligned. Each row width = 2*row + 1, base is widest.

        Returns:
            str: Multi-line string representing the pyramid.
        """
        self._validate_dim(height, "Height")
        self._validate_symbol(symbol)
        lines = []
        for row in range(height):
            symbols_in_row = 2 * row + 1
            spaces = height - row - 1
            line = ' ' * spaces + symbol * symbols_in_row + ' ' * spaces
            lines.append(line)
        return '\n'.join(lines)

# Example usage and rudimentary testability
if __name__ == "__main__":
    art = AsciiArt()
    print("Square:\n", art.draw_square(4, '*'), "\n")
    print("Rectangle:\n", art.draw_rectangle(6, 3, '#'), "\n")
    print("Parallelogram:\n", art.draw_parallelogram(5, 4, '+'), "\n")
    print("Triangle:\n", art.draw_triangle(5, 4, '!'), "\n")
    print("Pyramid:\n", art.draw_pyramid(4, '@'), "\n")
```
